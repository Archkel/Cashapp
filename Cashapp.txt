<script>
let fakeAccounts = {};

function login() {
    const username = document.getElementById('username').value;
    const password = document.getElementById('password').value;

    if(username === "" || password === "") {
        alert("Please enter both username and password!");
        return;
    }

    if(!fakeAccounts[username]) {
        fakeAccounts[username] = { password: password, balance: 0 };
    }

    document.getElementById('user-name').innerText = username;
    document.getElementById('balance').innerText = `Balance: $${fakeAccounts[username].balance.toFixed(2)}`;
    
    document.getElementById('login').style.display = "none";
    document.getElementById('dashboard').style.display = "block";
    document.getElementById('logout-btn').style.display = "block";
}

function generateMoney() {
    const username = document.getElementById('user-name').innerText;
    const amount = parseFloat(document.getElementById('amount').value);

    if(isNaN(amount) || amount <= 0) {
        alert("Enter a valid amount!");
        return;
    }

    fakeAccounts[username].balance += amount;
    document.getElementById('balance').innerText = `Balance: $${fakeAccounts[username].balance.toFixed(2)}`;
    document.getElementById('amount').value = "";
    alert(`$${amount} added! (FAKE)`);
}

function logout() {
    document.getElementById('login').style.display = "block";
    document.getElementById('dashboard').style.display = "none";
    document.getElementById('logout-btn').style.display = "none";
    document.getElementById('username').value = "";
    document.getElementById('password').value = "";
}
</script>
